function greet(language) {
  return function (name) {
    if (language === "en") {
      console.log("Hello, " + name);
    } else if (language === "es") {
      console.log("Hola, " + name);
    }
  };
}

const greetEnglish = greet("en");
const greetSpanish = greet("es");

greetEnglish("Aman");
greetSpanish("Ravi");
